 54  tower = Tower(10, None, 10, 10, "rectprism") 
 55  done = False 
 56  reflections = 0 
 57  while not done and 'y'==raw_input("continue?"): 
 58      #get a collision 
 59      record = tower.get_record(photon) 
 60      #check 3 things 
 61      #exiting 
 62      if record.is_exiting: 
 63          done = True  # move onto next photon 
 64      #wrap around 
 65      elif record.is_boundary: 
 66          photon.wrap_around(record) 
 67          print("wrapped around, info is",photon.position, photon.velocity) 
 68      #absorbed 
 69      elif random() < .1:
